CREATE TABLE Handlowcy 
    ( 
     Pesel             CHAR (11 CHAR)  NOT NULL , 
     Imię              VARCHAR2 (60 CHAR)  NOT NULL , 
     Nazwisko          VARCHAR2 (60 CHAR)  NOT NULL , 
     Stanowisko        VARCHAR2 (60 CHAR)  NOT NULL , 
     Data_zatrudnienia DATE  NOT NULL , 
     Prowizja          NUMBER (4,2)  NOT NULL , 
     CONSTRAINT chk_handlowcy_stanowisko CHECK (Stanowisko IN ('Starszy handlowiec', 'Handlowiec')) ,
     CONSTRAINT chk_handlowcy_prowizja CHECK (Prowizja >= 0) ,
     CONSTRAINT chk_handlowcy_pesel CHECK (REGEXP_LIKE(Pesel, '^[0-9]{11}$'))
    ) 
;

ALTER TABLE Handlowcy 
    ADD CONSTRAINT Handlowiec_PK PRIMARY KEY ( Pesel ) ;

CREATE TABLE Kierownicy 
    ( 
     Pesel             CHAR (11 CHAR)  NOT NULL , 
     Imię              VARCHAR2 (60 CHAR)  NOT NULL , 
     Nazwisko          VARCHAR2 (60 CHAR)  NOT NULL , 
     Stanowisko        VARCHAR2 (60 CHAR)  NOT NULL , 
     Data_zatrudnienia DATE  NOT NULL , 
     Zysk_z_projektu   NUMBER (4,2)  NOT NULL ,
     CONSTRAINT chk_kierownicy_stanowisko CHECK (Stanowisko IN ('Kierownik projektu', 'Zastępca kierownika')) ,
     CONSTRAINT chk_kierownicy_zysk_z_projektu CHECK (Zysk_z_projektu > 0) ,
     CONSTRAINT chk_kierownicy_pesel CHECK (REGEXP_LIKE(Pesel, '^[0-9]{11}$')) 
    ) 
;

ALTER TABLE Kierownicy 
    ADD CONSTRAINT Kierownik_PK PRIMARY KEY ( Pesel ) ;

CREATE TABLE Klienty 
    ( 
     NIP              CHAR (10 CHAR)  NOT NULL , 
     Nazwa_firmy      VARCHAR2 (100 CHAR)  NOT NULL , 
     Osoba_kontaktowa VARCHAR2 (100 CHAR)  NOT NULL , 
     Telefon          NUMBER (9)  NOT NULL , 
     Mail             VARCHAR2 (320) ,
     CONSTRAINT chk_klienty_nip CHECK (REGEXP_LIKE(NIP, '^[0-9]{10}$')) ,
     CONSTRAINT chk_klienty_mail CHECK (REGEXP_LIKE(Mail,'^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$')) ,
     CONSTRAINT klienty_mail_uq UNIQUE (Mail)
    )
;

ALTER TABLE Klienty 
    ADD CONSTRAINT Klient_PK PRIMARY KEY ( NIP ) ;

CREATE TABLE Lokalizacje 
    ( 
     Ulica        VARCHAR2 (50 CHAR)  NOT NULL , 
     Numer_domu   VARCHAR2 (5)  NOT NULL , 
     Kod_pocztowy VARCHAR2 (6)  NOT NULL , 
     Numer_Lokalu NUMBER (4)  NOT NULL , 
     Miasto       VARCHAR2 (30 CHAR)  NOT NULL , 
     Kraj         VARCHAR2 (30 CHAR)  NOT NULL ,
     CONSTRAINT chk_lokalizacje_numer_domu CHECK (REGEXP_LIKE(Numer_domu,'^[0-9]+$')) ,
     CONSTRAINT chk_lokalizacje_kod_pocztowy CHECK (REGEXP_LIKE(kod_pocztowy, '^[0-9]{2}-[0-9]{3}$')) ,
     CONSTRAINT chk_lokalizacje_numer_lokalu CHECK (Numer_lokalu >= 0) 
    ) 
;

ALTER TABLE Lokalizacje 
    ADD CONSTRAINT Lokalizacja_PK PRIMARY KEY ( Ulica, Numer_domu, Kod_pocztowy, Numer_Lokalu ) ;

CREATE TABLE Oferty 
    ( 
     Numer            VARCHAR2 (10)  NOT NULL ,

CREATE OR REPLACE TRIGGER trg_oferty_data
BEFORE INSERT OR UPDATE ON Oferty
FOR EACH ROW
BEGIN
    IF :NEW.Data_utworzenia > SYSDATE THEN
        RAISE_APPLICATION_ERROR(-20001, 'Data utworzenia nie może być z przyszłości');
    END IF;
END;
/

CREATE OR REPLACE TRIGGER trg_usterki_data
BEFORE INSERT OR UPDATE ON Usterki
FOR EACH ROW

        RAISE_APPLICATION_ERROR(-20002, 'Data wystąpienia nie może być z przyszłości');
    END IF;
END;
/
